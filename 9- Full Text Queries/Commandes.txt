  "query":{
    "match":{
      "title":"Developpement is a great job"
    }
  }
}

# Recherche suivant l'ordre des mots dans une phrase
GET /user/_search
{
  "query":{
    "match_phrase":{
      "title":"great developper"
    }
  }
}

# Recherche sur plusieurs champs
GET /user/_search
{
  "query":{
    "multi_match":{
      "query":"developper",
      "fields": ["title","description"]
    }
  }
}
